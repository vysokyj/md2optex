% book — fiction / prose book
% Pagella (Palatino-like), B5, symmetric two-sided margins, folio in header.
\fontfam[Pagella]
\typosize[11/13.5]
\margins/2 b5 (25,20,25,25)mm

% Czech typographic text composition
\widowpenalty=10000  \clubpenalty=10000
\frenchspacing
\emergencystretch=3em

% Empty defaults so the headline works even without book metadata.
\gdef\thetitle{}
\gdef\theauthor{}

% Running headers: even pages — folio + book title; odd pages — chapter title + folio.
% Footer empty (folio moved to header). Front-matter pages override locally via \bgroup.
\headline={\ifodd\pageno
  {\it\botmark}\hfil\folio
\else
  \folio\hfil{\it\thetitle}%
\fi}
\footline={}

% Chapter: openright (ensure odd page), centred title, thin-rule ornament, no chapter number.
% \_printrefnum[@] is invisible when \nonum is active but still writes the TOC entry
% and hyperlink anchor to the .ref file.
\_def\_printchap #1{\_vfill\_supereject
   % openright: insert blank recto page if the chapter would land on an even page
   \_ifodd\_pageno \_else
      \bgroup\footline={}\headline={}\null\vfil\_supereject\egroup
   \_fi
   \_prevdepth=0pt
   \_vglue3\_bigskipamount
   {\_chapfont \_leftskip=0pt plus1fill \_rightskip=\_leftskip
    \_noindent \_printrefnum[@]#1\_nbpar}%
   \_nobreak
   \vskip6pt
   {\noindent\hfil\vrule height0.4pt depth0pt width3em\hfil\par}%
   \_mark{#1}%
   \_nobreak\vskip\_bigskipamount\_nobreak
   \_firstnoindent
}

% Section: no number, left-aligned bold.
% \_insertmark omitted so the running header keeps the chapter title, not section title.
\_def\_printsec#1{\_par
   \_abovetitle{\_penalty-151}\_bigskip
   {\_secfont \_noindent \_raggedright \_printrefnum[@\_quad]#1\_nbpar}%
   \_nobreak \_belowtitle{\_medskip}%
   \_firstnoindent
}

% "Z praxe" callout — left rule with italic text, visually distinct from body text.
% Used for humorous or practical commentary that complements but doesn't interrupt the manual.
% Markdown syntax: fenced code block with language "praxe"
\def\zpraxe#1{\par\medskip
  \noindent\hbox{%
    \vrule width2.5pt%
    \hskip8pt%
    \vbox{\advance\hsize by-10.5pt\relax
      \parindent=0pt
      \smallskip{\it #1}\par\smallskip}}%
  \par\medskip}

% Subsection: no number, smaller bold
\_def\_printsecc#1{\_par
   \_abovetitle{\_penalty-101}{\_medskip\_smallskip}
   {\_seccfont \_noindent \_raggedright \_printrefnum[@\_quad]#1\_nbpar}%
   \_nobreak \_belowtitle{\_medskip}%
   \_firstnoindent
}
