% book — fiction / prose book
% Pagella (Palatino-like), B5, symmetric two-sided margins, folio centred in footer.
\fontfam[Pagella]
\typosize[11/13.5]
\margins/2 b5 (25,20,25,25)mm

% No page numbers in header, folio centred in footer
\headline={}
\footline={\hss\folio\hss}

% Chapter: page break, centred title, no "Chapter N" prefix
\_def\_printchap #1{\_vfill\_supereject \_prevdepth=0pt
   \_vglue3\_bigskipamount
   {\_chapfont \_leftskip=0pt plus1fill \_rightskip=\_leftskip
    \_noindent #1\_nbpar}\_mark{}%
   \_nobreak \_belowtitle{\_bigskip}%
   \_firstnoindent
}

% Section: no number, left-aligned bold
\_def\_printsec#1{\_par
   \_abovetitle{\_penalty-151}\_bigskip
   {\_secfont \_noindent \_raggedright #1\_nbpar}\_insertmark{#1}%
   \_nobreak \_belowtitle{\_medskip}%
   \_firstnoindent
}

% Subsection: no number, smaller bold
\_def\_printsecc#1{\_par
   \_abovetitle{\_penalty-101}{\_medskip\_smallskip}
   {\_seccfont \_noindent \_raggedright #1\_nbpar}%
   \_nobreak \_belowtitle{\_medskip}%
   \_firstnoindent
}
